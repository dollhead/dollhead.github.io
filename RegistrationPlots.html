dates = pd.read_csv("auth_user.csv", names = ['date'],parse_dates = True)
temp = []
for day in dates.date:
    index = day.index(' ')
    day = day[:index]
    temp.append(day)
dates.date = temp

q = """
select date,count(*) as registered
from dates
group by date
"""
reg_by_date = pysqldf(q)
dates_series = pd.DatetimeIndex(reg_by_date.date)
reg_by_date['cumulsum'] = np.cumsum(reg_by_date.registered)
